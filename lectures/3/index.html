Lab: construção da classe Crawler

Lab: construção da classe Crawler

Algoritmo do crawler

crawlWeb(QUrl seed) {
    fila.enqueue(seed)
    while (fila) {
        url = fila.dequeue()
        if (ja_baixou(url))
            continue;

        getPage(url, conteudo)            // implementar
        addPageToIndex(url, conteudo)
        getAllLinks(conteudo, linkslist)  // implementar getNextUrl()
        addLinksToGraph(url, linkslist)
        fila.enqueue(linkslist)
        ja_baixou.append(url)
    }
}
